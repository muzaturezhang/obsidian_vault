## 0. 先明确“clip”可能是什么

在你们这个项目里，clip 不一定等于一个 `.mp4` 文件。它可能是：

```text
一个 episode 的一个 player 视角
一个固定长度的视频片段
一个 manifest 里的一个 sample entry
一组同 basename 的文件：
xxx.mp4
xxx_depth.mkv
xxx_hud.mkv
xxx_motion.bin
xxx.json
xxx_episode_info.json
```

比如这个 basename：

```text
Ep_000010_team_2_player_0000_inst_000
```

可能对应一个 clip。

所以你统计时要避免只数 `.mp4`，更应该按 **basename / episode / player / map / duration** 来统计。

## 可直接用的脚本版本

```python
#!/usr/bin/env python3
import argparse
import csv
import json
import os
import re
import subprocess
from collections import defaultdict, Counter
from pathlib import Path


VIDEO_EXTS = {".mp4", ".mkv"}
KNOWN_SUFFIXES = [
    "_depth",
    "_hud",
]


def strip_known_suffix(path: Path) -> str:
    stem = path.stem
    for suffix in KNOWN_SUFFIXES:
        if stem.endswith(suffix):
            return stem[: -len(suffix)]
    if stem.endswith("_episode_info"):
        return stem[: -len("_episode_info")]
    return stem


def parse_basename(base: str) -> dict:
    """
    Example:
    Ep_000010_team_2_player_0000_inst_000
    """
    info = {
        "episode": "",
        "team": "",
        "player": "",
        "inst": "",
    }

    m = re.search(
        r"Ep_(?P<episode>\d+)_team_(?P<team>\d+)_player_(?P<player>\d+)_inst_(?P<inst>\d+)",
        base,
    )
    if m:
        info.update(m.groupdict())

    return info


def file_size_mb(p):
    if not p:
        return ""
    try:
        return round(os.path.getsize(p) / 1024 / 1024, 3)
    except OSError:
        return ""


def ffprobe_video(path: str) -> dict:
    """
    Read video metadata using ffprobe if available.
    Safe but can be slow on huge datasets, so call only for RGB mp4 by default.
    """
    empty = {
        "duration_sec": "",
        "width": "",
        "height": "",
        "fps": "",
        "num_frames": "",
    }

    if not path:
        return empty

    cmd = [
        "ffprobe",
        "-v", "error",
        "-select_streams", "v:0",
        "-show_entries",
        "stream=width,height,r_frame_rate,nb_frames:format=duration",
        "-of", "json",
        path,
    ]

    try:
        out = subprocess.check_output(cmd, stderr=subprocess.DEVNULL, timeout=10)
        obj = json.loads(out.decode("utf-8", errors="ignore"))

        stream = obj.get("streams", [{}])[0]
        fmt = obj.get("format", {})

        fps_raw = stream.get("r_frame_rate", "")
        fps = ""
        if fps_raw and "/" in fps_raw:
            a, b = fps_raw.split("/")
            try:
                fps = round(float(a) / float(b), 3) if float(b) != 0 else ""
            except Exception:
                fps = fps_raw

        duration = fmt.get("duration", "")
        if duration != "":
            try:
                duration = round(float(duration), 3)
            except Exception:
                pass

        return {
            "duration_sec": duration,
            "width": stream.get("width", ""),
            "height": stream.get("height", ""),
            "fps": fps,
            "num_frames": stream.get("nb_frames", ""),
        }

    except Exception:
        return empty


def read_episode_info(path: str) -> dict:
    """
    Try to extract map name or other metadata from *_episode_info.json.
    """
    result = {
        "map": "",
    }

    if not path:
        return result

    try:
        with open(path, "r", encoding="utf-8") as f:
            obj = json.load(f)

        # Try common names.
        for key in ["map", "map_name", "mapName", "level", "scene"]:
            if key in obj:
                result["map"] = obj[key]
                break

        # If nested, do a shallow search.
        if not result["map"]:
            for v in obj.values():
                if isinstance(v, dict):
                    for key in ["map", "map_name", "mapName", "level", "scene"]:
                        if key in v:
                            result["map"] = v[key]
                            break
                if result["map"]:
                    break

    except Exception:
        pass

    return result


def scan_roots(roots, probe_video=False):
    groups = defaultdict(dict)

    for root in roots:
        root = Path(root)
        if not root.exists():
            print(f"[WARN] root does not exist: {root}")
            continue

        for p in root.rglob("*"):
            if not p.is_file():
                continue

            base = strip_known_suffix(p)
            key = (str(root), str(p.parent), base)
            ext = p.suffix.lower()
            name = p.name

            if ext == ".mp4":
                groups[key]["rgb_path"] = str(p)
            elif ext == ".mkv" and p.stem.endswith("_depth"):
                groups[key]["depth_path"] = str(p)
            elif ext == ".mkv" and p.stem.endswith("_hud"):
                groups[key]["hud_path"] = str(p)
            elif ext == ".bin" and p.stem.endswith("_motion"):
                groups[key]["motion_path"] = str(p)
            elif ext == ".json" and p.stem.endswith("_episode_info"):
                groups[key]["episode_info_path"] = str(p)
            elif ext == ".json":
                groups[key]["json_path"] = str(p)
            else:
                groups[key].setdefault("other_files", []).append(str(p))

    rows = []
    for (root, parent, base), files in groups.items():
        parsed = parse_basename(base)

        rgb_path = files.get("rgb_path", "")
        depth_path = files.get("depth_path", "")
        hud_path = files.get("hud_path", "")
        motion_path = files.get("motion_path", "")
        json_path = files.get("json_path", "")
        episode_info_path = files.get("episode_info_path", "")

        episode_info = read_episode_info(episode_info_path)
        video_info = ffprobe_video(rgb_path) if probe_video else {
            "duration_sec": "",
            "width": "",
            "height": "",
            "fps": "",
            "num_frames": "",
        }

        row = {
            "clip_id": base,
            "root": root,
            "dir": parent,
            "basename": base,
            "episode": parsed["episode"],
            "map": episode_info["map"],
            "team": parsed["team"],
            "player": parsed["player"],
            "inst": parsed["inst"],

            "has_rgb": bool(rgb_path),
            "has_depth": bool(depth_path),
            "has_hud": bool(hud_path),
            "has_motion": bool(motion_path),
            "has_json": bool(json_path),
            "has_episode_info": bool(episode_info_path),

            "rgb_path": rgb_path,
            "depth_path": depth_path,
            "hud_path": hud_path,
            "motion_path": motion_path,
            "json_path": json_path,
            "episode_info_path": episode_info_path,

            "rgb_size_mb": file_size_mb(rgb_path),
            "depth_size_mb": file_size_mb(depth_path),
            "hud_size_mb": file_size_mb(hud_path),
            "motion_size_mb": file_size_mb(motion_path),
        }

        row.update(video_info)
        rows.append(row)

    return rows


def summarize(rows):
    summary = {}

    summary["num_clip_groups"] = len(rows)
    summary["num_with_rgb"] = sum(r["has_rgb"] for r in rows)
    summary["num_with_depth"] = sum(r["has_depth"] for r in rows)
    summary["num_with_hud"] = sum(r["has_hud"] for r in rows)
    summary["num_with_motion"] = sum(r["has_motion"] for r in rows)
    summary["num_with_json"] = sum(r["has_json"] for r in rows)
    summary["num_with_episode_info"] = sum(r["has_episode_info"] for r in rows)

    summary["roots"] = dict(Counter(r["root"] for r in rows))
    summary["maps"] = dict(Counter(r["map"] or "UNKNOWN" for r in rows))
    summary["episodes"] = len(set(r["episode"] for r in rows if r["episode"]))
    summary["players"] = len(set((r["team"], r["player"]) for r in rows if r["player"]))

    complete_rows = [
        r for r in rows
        if r["has_rgb"] and r["has_depth"] and r["has_hud"] and r["has_motion"] and r["has_json"] and r["has_episode_info"]
    ]
    summary["num_complete_multimodal_groups"] = len(complete_rows)

    durations = []
    for r in rows:
        try:
            if r["duration_sec"] != "":
                durations.append(float(r["duration_sec"]))
        except Exception:
            pass

    if durations:
        summary["duration_sec"] = {
            "min": min(durations),
            "max": max(durations),
            "avg": sum(durations) / len(durations),
        }

    return summary


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument(
        "--roots",
        nargs="+",
        default=[
            "/mnt/data/csgo-datasets",
            "/mnt/data/csgo-datasets-fullsubset",
        ],
    )
    parser.add_argument("--out_csv", default="outputs/csgo_metadata.csv")
    parser.add_argument("--out_json", default="outputs/csgo_metadata_summary.json")
    parser.add_argument(
        "--probe_video",
        action="store_true",
        help="Use ffprobe to read video duration/fps/resolution. Slower.",
    )
    args = parser.parse_args()

    rows = scan_roots(args.roots, probe_video=args.probe_video)

    Path(args.out_csv).parent.mkdir(parents=True, exist_ok=True)
    Path(args.out_json).parent.mkdir(parents=True, exist_ok=True)

    if rows:
        fieldnames = list(rows[0].keys())
        with open(args.out_csv, "w", newline="", encoding="utf-8") as f:
            writer = csv.DictWriter(f, fieldnames=fieldnames)
            writer.writeheader()
            writer.writerows(rows)

    summary = summarize(rows)
    with open(args.out_json, "w", encoding="utf-8") as f:
        json.dump(summary, f, indent=2, ensure_ascii=False)

    print(f"[OK] wrote csv: {args.out_csv}")
    print(f"[OK] wrote json: {args.out_json}")
    print(json.dumps(summary, indent=2, ensure_ascii=False))


if __name__ == "__main__":
    main()
```
